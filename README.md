	allprojects {<br>
		repositories {<br>
			...<br>
			maven { url 'https://jitpack.io' }<br>
		}<br>
	}<br> 
  <br>
  Step 2. Add the dependency <br>
  dependencies { <br>
	        compile 'com.github.FantasyEngineer:BaseApp:v1.0' <br>
	} <br>
